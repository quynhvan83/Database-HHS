#<Start the script>
{
	tmp = GetProperty(LF_MSI_CHK_CHAR_TXMIT)

	# If the scanner property of Transmit MSI check character(s) = Don’t transmit check character(s) , the returned value will be:
	# NO TRANSMIT CHECK CHAR – Label Data
	if (tmp == 0)
	{
	       StripBeginEndChars(OUT1,0,1)
              OUT1.Data = "NO TRANSMIT CHECK CHAR - " + OUT1.Data + "\r\n"
	}

	# If the scanner property of Transmit MSI check character(s) = Transmit check character(s), the returned value will be:
	# TRANSMIT CHECK CHAR- Label Data
	else if (tmp == 1)
	{
		OUT1.Data = "TRANSMIT CHECK CHAR - " + OUT1.Data + "\r\n"
	}
DoSLF(OUT1)
}
#<End the script>
