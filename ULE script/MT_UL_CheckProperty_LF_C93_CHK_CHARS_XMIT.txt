#<Start the script>
{
	tmp = GetProperty(LF_C93_CHK_CHARS_XMIT)

	# If the scanner property of Transmit Code 93 check characters = Don’t transmit check characters , the returned value will be:
	# NO TRANSMIT CHECK CHAR – Label Data
	if (tmp == 0)
	{
		OUT1.Data = "NO TRANSMIT CHECK CHAR - " + OUT1.Data
	}

	# If the scanner property of Transmit Code 93 check characters = Transmit check characters, the returned value will be:
	# TRANSMIT CHECK CHAR- Label Data
	else if (tmp == 1)
	{
		OUT1.Data = "TRANSMIT CHECK CHAR - " + OUT1.Data
	}
	DoSLF(OUT1)
}
#<End the script>
