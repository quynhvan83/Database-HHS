{
	tmp = GetProperty(LF_MSI_CHK_CHAR_CALC)

	# If the scanner property of Perform MSI check character check = No check character check , the returned value will be:
	# NO PERFORM CHECK CHAR â€“ Label Data
	if (tmp == 0)
	{
		OUT1.Data = "NO PERFORM CHECK CHAR - " + OUT1.Data + "\r\n"
	}

	# If the scanner property of Perform MSI check character check = Do check character check, the returned value will be:
	# PERFORM CHECK CHAR- Label Data
	else if (tmp == 1)
	{
		OUT1.Data = "PERFORM CHECK CHAR - " + OUT1.Data + "\r\n"
	}
DoSLF(OUT1)
}
