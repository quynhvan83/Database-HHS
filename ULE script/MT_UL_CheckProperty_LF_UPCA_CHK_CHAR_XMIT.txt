#<Start the script>
{

	# <If Data Type = UPCA: Start>
	if (OUT1.DataType == "UPCA")
	{
	
	# Get Property of UPCE Check Character Transmission:
	upca_CT = GetProperty(LF_UPCA_CHK_CHAR_XMIT)

		# If the scanner property = 0, the returned value will be:
		# NO UPCA CHECK CHARACTER TRANSMISSION – Label Data
		if (upca_CT == 0)
		{
			# Strip Check Character
			StripBeginEndChars(OUT1,0,1)
			# Data will displays with no Check Characters Transmission
			OUT1.Data = "NO CHECK CHARACTER TRANSMISSION - " + OUT1.Data + "\r"
		}

		# If the scanner property = 1, the returned value will be:
		# UPCA CHECK CHARACTER TRANSMISSION – Label Data
		else if (upca_CT == 1)
		{
			OUT1.Data = "UPCA CHECK CHARACTER TRANSMISSION - " + OUT1.Data + "\r"
		}
	}
	# <If Data Type = UPCA: End>
}
#<End the script>
