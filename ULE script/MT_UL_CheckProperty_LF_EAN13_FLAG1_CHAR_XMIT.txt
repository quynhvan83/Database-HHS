#<Start the script>
{

	# <If Data Type = EAN13: Start>
	if (OUT1.DataType == "EAN13")
	{
	
	# Get Property of EAN13 Flag1 Character Transmission:
	ean13_flag1 = GetProperty(LF_EAN13_FLAG1_CHAR_XMIT)

		# If the scanner property = 0, the returned value will be:
		# NO EAN13 FLAG CHARACTER TRANSMISSION – Label Data
		if (ean13_flag1 == 0)
		{
			# Strip Check Character
			StripBeginEndChars(OUT1,1,0)
			# Data will displays with no Flag1 Characters Transmission
			OUT1.Data = "NO EAN13 FLAG1 CHARACTER TRANSMISSION - " + OUT1.Data + "\r"
		}

		# If the scanner property = 1, the returned value will be:
		# EAN13 CHECK CHARACTER TRANSMISSION – Label Data
		else if (ean13_flag1 == 1)
		{
			OUT1.Data = "EAN13 FLAG1 CHARACTER TRANSMISSION - " + OUT1.Data + "\r"
		}
	}
	# <If Data Type = EAN13: End>
}
#<End the script>
