#<Start the script>
{
	tmp = GetProperty(LF_C128_CHK_CHAR_XMIT)

	# If the scanner property of Transmit Code 128 check character = Don’t transmit check character , the returned value will be:
	# NO TRANSMIT CHECK CHAR – Label Data
	if (tmp == 0)
	{
		OUT1.Data = "NO TRANSMIT CHECK  CHAR - " + OUT1.Data + "\r\n"
	}

	# If the scanner property of Transmit Code 128 check character = transmit check character, the returned value will be:
	# TRANSMIT CHECK CHAR- Label Data
	else if (tmp == 1)
	{
		OUT1.Data = "TRANSMIT CHECK CHAR - " + OUT1.Data + "\r\n"
	}
DoSLF(OUT1)
}
#<End the script>
