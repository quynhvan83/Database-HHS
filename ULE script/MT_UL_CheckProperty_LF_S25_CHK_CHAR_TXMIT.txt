#<Start the script>
{

	# <If Data Type = Standard 2 of 5: Start>
	if (OUT1.DataType == "S2of5")
	{

	# Get Property of Standard 2 of 5 Check Character Calculation:
	s2of5_ct = GetProperty(LF_S25_CHK_CHAR_TXMIT)

		# If the scanner property = 0, the returned value will be:
		# NO S2OF5 CHECK CHAR TRANSMISSION – Label Data
		if (s2of5_ct == 0)
		{
			# Strip Check Character
			StripBeginEndChars(OUT1,0,1)
			# Data will displays with NO S2OF5 CHECK CHAR TRANSMISSION:
			OUT1.Data = "NO S2OF5 CHECK CHAR TRANSMISSION - " + OUT1.Data + "\r\n"
		}

		# If the scanner property = 1, the returned value will be:
		# S2OF5 CHECK CHAR TRANSMISSION – Label Data
		else if (s2of5_ct == 1)
		{
			# Data will displays: S2OF5 CHECK CHAR TRANSMISSION
			OUT1.Data = "S2OF5 CHECK CHAR TRANSMISSION - " + OUT1.Data + "\r\n"
		}
	DoSLF(OUT1)
	}
	# <If Data Type = Standard 2 of 5: End>

}
#<End the script>
